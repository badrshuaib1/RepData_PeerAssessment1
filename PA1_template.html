Imputing missing values

impute_function<- function(df, values){

        for (i in 1:nrow(df)){
                if (is.na(df[i,1])) {
                        df[i,1]<- values[grep(paste("^",df1[i,3],"$", sep= ""), values[,1]),2]

                        }
        }
       return(df)
}
new_df<- impute_function(df1, intervalmean)
